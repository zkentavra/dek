1.	За допомого мови запитів SQL в базі даних "Студент", яка знаходиться за адресою D:\SQL,встановити зв’язки між таблицями, необхідними для знаходження номера телефону студентів спеціальності «Обслуговування програмних систем і комплексів», в яких діда звати Петро.
use Студент
SELECT DISTINCT ПРІЗВИЩЕ,ТЕЛЕФОН,повна_назва, піб_батька,піб_матері
FROM студенти$,спеціальність$,контактна_інформація$
WHERE  студенти$.код_студ = контактна_інформація$.код_студ AND спеціальність$.повна_назва LIKE '%систем%' AND контактна_інформація$.піб_батька LIKE '%Петрович%' 
OR студенти$.код_студ = контактна_інформація$.код_студ AND спеціальність$.повна_назва LIKE '%систем%' AND контактна_інформація$.піб_матері LIKE '%Петрівна%' ;
